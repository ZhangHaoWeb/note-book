var largestRectangleArea = function(heights) {
    let ans = 0

    // å·¦è¾¹ç•Œ
    for (let left = 0; left < heights.length; left++) {
        let h = heights[left];
        // å³è¾¹ç•Œ
        for (let right = left; right < heights.length; right++) {
            h = Math.min(h, heights[right])

            ans = Math.max(ans, w * h)
        }
    }
    return ans
};
