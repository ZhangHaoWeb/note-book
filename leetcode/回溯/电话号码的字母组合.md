/**
 * @param {string} digits
 * @return {string[]}
 */
var letterCombinations = function(digits) {
    if(digits.length <= 0) {
        return []
    }
    
    const result = []
    const map = new Map([
        ["2", ["a", "b", "c"]],
        ["3", ["d", "e", "f"]],
        ["4", ["g", "h", "i"]],
        ["5", ["j", "k", "l"]],
        ["6", ["m", "n", "o"]],
        ["7", ["p", "q", "r", "s"]],
        ["8", ["t", "u", "v"]],
        ["9", ["w", "x", "y", "z"]],
    ])
    // çœŸæƒ³æš´åŠ›å•ŠğŸ˜ˆ
    
    function backtrack(index, path) {
        if (index == digits.length) {
            result.push(path.join(""))
            return
        }
        let digit = digits[index]
        let arr = map.get(digit)
        for (let i = 0; i < arr.length; i++) {
            path.push(arr[i])
            backtrack(index + 1, path)
            path.pop()
        }
    }

    backtrack(0, [])
    return result
};
